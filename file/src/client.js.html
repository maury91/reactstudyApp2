    function getTimers(opts) {
    $.ajax({ // eslint-disable-line no-undef
        url: '/api/timers',
        data: null,
        method: 'get',
        dataType: 'json',
        cache: false,
        success: opts.success,
        error: function (xhr, status, err) {
            console.error('GET /api/timers', status, err.toString()); // eslint-disable-line no-console
        },
    });
}
